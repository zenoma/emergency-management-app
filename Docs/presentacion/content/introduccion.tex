\section{Introducción}

\begin{frame}[c,shrink]{Contexto y motivación}

\begin{block}{Galicia: elevada exposición a emergencias}
    \begin{itemize}
        \item Incendios forestales: más de 1.000 anuales en la última década, con más de 10.000 ha afectadas.
        \item Inundaciones derivadas de episodios de lluvia intensa.
        \item Derrumbes de carreteras y terraplenes.
        \item Riesgos industriales.
    \end{itemize}
\end{block}

\begin{block}{Problemas detectados}
    \begin{itemize}
        \item Fragmentación de herramientas y sistemas entre organismos.
        \item Dificultad para coordinar múltiples actores en contextos de alta presión.
        \item Escasa integración de datos y fuentes de información.
    \end{itemize}
\end{block}

\end{frame}
