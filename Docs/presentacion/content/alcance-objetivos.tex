\section{Alcance y Objetivos}

\begin{frame}[c,shrink]{Alcance y Objetivos}

    \begin{block}{Alcance}
        \setbeamertemplate{itemize items}{}
            \begin{itemize}
                \item Plataforma para la gestión integrada de emergencias multi-riesgo en Galicia, combinando una plataforma web evolucionada y una aplicación móvil creada desde cero.
            \end{itemize}
        \setbeamertemplate{itemize items}[default]
    \end{block}

    \begin{block}{Objetivos principales}
        \begin{itemize}
            \item Gestión de alertas ciudadanas en tiempo real mediante fotografías geolocalizadas.
            \item Recomendación semiautomatizada de planes de actuación según tipo y nivel de riesgo.
            \item Coordinación de recursos de múltiples organizaciones con inventario dinámico.
            \item Notificaciones push para comunicación de emergencias y avisos.
            \item Mapa interactivo con emergencias, cuadrantes y recursos en tiempo real.
            \item Análisis post-emergencia con registros detallados y trazabilidad.
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}[c,shrink]{Del TFG al TFM}

\begin{block}{Punto de partida: TFG}
    \begin{itemize}
        \item Aplicación web para coordinación de brigadas en extinción de incendios forestales.
        \item Gestión de recursos, visualización de cuadrantes, seguimiento de actuaciones.
    \end{itemize}
\end{block}

\begin{block}{Evolución: TFM}
    \begin{itemize}
        \item Plataforma multi-riesgo: incendios, inundaciones, derrumbes, riesgos industriales.
        \item Nueva aplicación móvil Android para trabajo en campo y participación ciudadana.
        \item Notificaciones push en tiempo real con Firebase Cloud Messaging.
        \item Recomendación semiautomática de recursos basada en reglas.
    \end{itemize}
\end{block}

\end{frame}

\begin{frame}[c,shrink]{Objetivos concretos}

    \begin{block}{Aplicación móvil}
        \begin{itemize}
            \item App Android con notificaciones push.
            \item Mapa con cuadrantes, eventos y recursos asignados.
            \item Subida de imágenes para avisos ciudadanos.
        \end{itemize}
    \end{block}

    \begin{block}{Plataforma web}
        \begin{itemize}
            \item Gestión de nuevos tipos de emergencia y organizaciones.
            \item Mapa interactivo con emergencias, recursos y su estado.
            \item Recomendación semiautomática de recursos basada en reglas.
            \item Modelo flexible para organizaciones, manteniendo soporte a brigadas forestales.
        \end{itemize}
    \end{block}

\end{frame}
