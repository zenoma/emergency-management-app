\section{Conclusiones y Trabajo Futuro}

    \begin{frame}[c,shrink]{Conclusiones}
        \begin{block}{Objetivos cumplidos}
            \begin{itemize}
                \item Plataforma multi-riesgo que integra incendios, inundaciones, derrumbes y riesgos industriales.
                \item App Android con notificaciones push, mapa interactivo y avisos geolocalizados.
                \item Coordinación de recursos mediante asignaciones dinámicas con ciclo de vida completo.
                \item Motor de recomendación semiautomática basado en reglas configurables.
                \item Trazabilidad operativa unificada mediante registros de asignaciones.
            \end{itemize}
        \end{block}
    \end{frame}

    \begin{frame}[c,shrink]{Lecciones aprendidas y Trabajo Futuro} 
        \begin{block}{Dificultades encontradas}
            \begin{itemize}
                \item Integración de Firebase Cloud Messaging.
                \item Diseño del motor de reglas de recomendación en JSONB.
            \end{itemize}
        \end{block}
        
        \begin{block}{Trabajo Futuro}
            \begin{itemize}
                \item Incorporación de inteligencia artificial generativa para predicción y recomendación dinámica.
                \item Seguimiento GPS continuo de recursos desplegados.
                \item Ampliación del dominio: temporales costeros, riesgos sísmicos, emergencias sanitarias.
            \end{itemize}
        \end{block}
    \end{frame}
